67  bytes fromHex(std::string const& _s)
68  {
69  unsigned s = (_s.size() >= 2 && _s[0] == '0' && _s[1] == 'x') ? 2 : 0;
70  std::vector<uint8_t> ret;
71  ret.reserve((_s.size() - s + 1) / 2);
72 
73  if (_s.size() % 2)
74  {
75  int h = fromHexChar(_s[s++]);
76  if (h != -1)
77  ret.push_back(h);
78  else
79  return bytes();
80  }
81  for (unsigned i = s; i < _s.size(); i += 2)
82  {
83  int h = fromHexChar(_s[i]);
84  int l = fromHexChar(_s[i + 1]);
85  if (h != -1 && l != -1)
86  ret.push_back((byte)(h * 16 + l));
87  else
88  return bytes();
89  }
90  return ret;
91  }

112  std::string toHex(Iterator _it, Iterator _end, std::string const& _prefix) {
113  typedef std::iterator_traits<Iterator> traits;
114  static_assert(sizeof(typename traits::value_type) == 1, "toHex needs byte-sized element type");

116  static char const* hexdigits = "0123456789abcdef";
117  size_t off = _prefix.size();
118  std::string hex(std::distance(_it, _end)*2 + off, '0');
119  hex.replace(0, off, _prefix);

121  {
122  hex[off++] = hexdigits[(*_it >> 4) & 0x0f];
123  hex[off++] = hexdigits[*_it & 0x0f];
